(DEFUN DIFF (X Y) (COND ((EQ X Y) NIL) (T T)))
(DEFUN MEM (X LIST)
(COND ( (NULL LIST) NIL ) ( T (COND
((EQ X (CAR LIST)) T) ( T (MEM X (CDR LIST)))))))

(DEFUN F (X) (PLUS X Y))
(DEFUN G (Y) (F 10))
(DEFUN H (Y) (F 20))
(G 5)
(H 5)
(G (H 5))
(MEM 5 (QUOTE (7 4 4)))
(MEM 5 (CONS 5 5))
(COND (NIL NIL) (NIL NIL) (T 7))
